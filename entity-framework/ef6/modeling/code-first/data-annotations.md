
<span data-ttu-id="38d90-138">При наличии сущностей с помощью составного внешние ключи, необходимо указать тот же столбец, упорядочение, которое использовалось для соответствующего свойства первичного ключа.</span><span class="sxs-lookup"><span data-stu-id="38d90-138">If you have entities with composite foreign keys, then you must specify the same column ordering that you used for the corresponding primary key properties.</span></span>

``` csharp
        public int StampId { get; set; }
        public DateTime Stamped { get; set; }
        public string StampingCountry { get; set; }

        [ForeignKey("Passport")]
        [Column(Order = 1)]
        public int PassportNumber { get; set; }

        [ForeignKey("Passport")]
```
